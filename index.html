app.component.ts 

                
export class AppComponent implements OnInit {
  speaker: Speaker;
  speakers: Speaker[] = [];

  ngOnInit() {
    this.initModel();
  }

  initModel() {
    this.speaker = new Speaker('', '');
  }

  addSpeaker(speaker) {
    this.speakers.push(speaker);
    this.initModel();
  }
}
